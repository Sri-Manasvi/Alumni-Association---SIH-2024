            const navMenu = document.getElementById('navMenu');

            navMenu.style.display = (navMenu.style.display === 'flex') ? 'none' : 'flex';

        }

 

        function navigate(sectionId) {

            const sections = ['homePage', 'masters', 'jobs', 'events', 'mentorship', 'feedback', 'achievements', 'resume'];

            sections.forEach(id => {

                const section = document.getElementById(id);

                if (section) {

                    section.style.display = (id === sectionId) ? 'block' : 'none';

                }

            });

        }

 

        function showLogin() {

            // Check if the login container already exists

            if (document.querySelector('.login-container')) {

                document.getElementById('login-container').style.display = 'flex';

                return; // Exit if already present

            }

 

            const loginContainer = document.createElement('div');

            loginContainer.className = 'login-container';

            loginContainer.id = 'login-container';

            loginContainer.innerHTML = `

                <h2>Login</h2>

                <input type="text" id="userId" class="input-field" placeholder="User ID">

                <input type="password" id="password" class="input-field" placeholder="Password">

                <button class="login-button" onclick="performLogin()">Login</button>

                <button class="login-button" onclick="hideLogin()">Cancel</button>

            `;

            document.body.appendChild(loginContainer);

            document.getElementById('login-container').style.display = 'flex';

        }

 

        function hideLogin() {

            const loginContainer = document.getElementById('login-container');

            if (loginContainer) {

                loginContainer.style.display = 'none';

            }

        }

 

        function performLogin() {

            const validUserIds = ['2023001234', '2023001533', '2023001354', '2023003987'];

            const validPassword = 'Gitam';

 

            const userId = document.getElementById('userId').value;

            const password = document.getElementById('password').value;

 

            if (validUserIds.includes(userId) && password === validPassword) {

                hideLogin();

                navigate('homePage');

            } else {

                alert('Invalid credentials, please try again.');

            }

        }

 

        function postFeedback() {

            const feedbackTextbox = document.getElementById('feedback-textbox');

            const feedbackComments = document.getElementById('feedback-comments');

 

            const newFeedback = feedbackTextbox.value.trim();

            if (newFeedback) {

                const newComment = document.createElement('p');

                newComment.textContent = `New User: "${newFeedback}"`;

                feedbackComments.appendChild(newComment);

                feedbackTextbox.value = ''; // Clear the textbox after posting

            } else {

                alert('Please enter your feedback before posting.');

            }

        }

    </script>

</body>

 

</html>
